EoF_debug_category = {
	EoF_show_school_gui = {
		icon = GFX_decision_generic_break_treaty
		visible = {
			NOT = {
				has_country_flag = EoF_show_school_window
			}
		}
		complete_effect = {
			set_country_flag = EoF_show_school_window
		}
	}

	EoF_add_to_index = {
		icon = GFX_decision_generic_break_treaty
		visible = {
			NOT = {
				set_temp_variable = {
					max_index = ROOT.EoF_army_leaders^num
				}
				subtract_from_temp_variable = {
					max_index = 1
				}
				check_variable = {
					var = ROOT.selected_leader_index_var
					value = max_index
					compare = greater_than_or_equals
				}
			}
		}
		complete_effect = {
			ROOT = {
				add_to_variable ={ selected_leader_index_var = 1 }
			}
		}
	}

	EoF_subtract_from_index = {
		icon = GFX_decision_generic_break_treaty
		visible = {
			NOT = {
				check_variable = {
					ROOT.selected_leader_index_var < 1
				}
			}
		}
		complete_effect = {
			ROOT = {
				subtract_from_variable = { selected_leader_index_var = 1 }
			}
		}
	}
}
