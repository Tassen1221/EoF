pick_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { poster_left_icon_frame = global.propaganda_empty_slot_index }
			}
			set_variable = { poster_left_icon_frame = new_propaganda_campaign }
		}
		else_if = {
			limit = {
				check_variable = { poster_center_icon_frame = global.propaganda_empty_slot_index }
			}
			set_variable = { poster_center_icon_frame = new_propaganda_campaign }
		}
		set_variable = { new_propaganda_campaign = 0 } # Reset var
		add_to_variable = { propaganda_campaigns_taken = 1 } #Update propaganda campaigns counter
	}
}

reset_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { poster_left_icon_frame = propaganda_campaign_to_remove }
			}
			set_variable = { poster_left_icon_frame = global.propaganda_empty_slot_index }
		}
		else_if = {
			limit = {
				check_variable = { poster_center_icon_frame = propaganda_campaign_to_remove }
			}
			set_variable = { poster_center_icon_frame = global.propaganda_empty_slot_index }
		}
		set_variable = { propaganda_campaign_to_remove = 0 } # Reset var
		subtract_from_variable = { propaganda_campaigns_taken = 1}
	}
}
