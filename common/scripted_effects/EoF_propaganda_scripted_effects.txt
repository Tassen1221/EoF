EQS_pick_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { EQS_poster_left_icon_frame = EQS_propaganda_empty_slot_index }
			}
			set_variable = { EQS_poster_left_icon_frame = EQS_new_propaganda_campaign }
			# #Ageing posters
			# set_variable = { EQS_propaganda_left_poster_to_age = EQS_poster_left_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
		}
		else_if = {
			limit = {
				check_variable = { EQS_poster_center_icon_frame = EQS_propaganda_empty_slot_index }
			}
			set_variable = { EQS_poster_center_icon_frame = EQS_new_propaganda_campaign }
			#Ageing posters
			# set_variable = { EQS_propaganda_center_poster_to_age = EQS_poster_center_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
		}
		set_variable = { EQS_new_propaganda_campaign = 0 } # Reset var
		add_to_variable = { EQS_propaganda_campaigns_taken = 1 } #Update propaganda campaigns counter
	}
}

EQS_reset_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { EQS_poster_left_icon_frame = EQS_propaganda_campaign_to_remove }
			}
			set_variable = { EQS_poster_left_icon_frame = EQS_propaganda_empty_slot_index }
			#Clear aged posters
			clear_variable = EQS_propaganda_left_poster_to_age
			clr_country_flag = EQS_propaganda_left_slot_is_old_flag
		}
		else_if = {
			limit = {
				check_variable = { EQS_poster_center_icon_frame = EQS_propaganda_campaign_to_remove }
			}
			set_variable = { EQS_poster_center_icon_frame = EQS_propaganda_empty_slot_index }
			#Clear aged posters
			# clear_variable = EQS_propaganda_center_poster_to_age
			# clr_country_flag = EQS_propaganda_center_slot_is_old_flag
		}
		set_variable = { EQS_propaganda_campaign_to_remove = 0 } # Reset var
		subtract_from_variable = { EQS_propaganda_campaigns_taken = 1}
	}
}
